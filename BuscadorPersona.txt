import { useState, useEffect, useRef } from 'react';
import { Search, Plus, X } from 'lucide-react';
import api from '../../services/api';

const BuscadorPersona = ({ value, onChange, onCrearNueva }) => {
  const [searchTerm, setSearchTerm] = useState('');
  const [resultados, setResultados] = useState([]);
  const [loading, setLoading] = useState(false);
  const [showResults, setShowResults] = useState(false);
  const [selectedPersona, setSelectedPersona] = useState(null);
  const wrapperRef = useRef(null);

  // Cargar persona seleccionada si hay value
  useEffect(() => {
    if (value && !selectedPersona) {
      cargarPersona(value);
    }
  }, [value]);

  // Cerrar resultados al hacer click fuera
  useEffect(() => {
    const handleClickOutside = (event) => {
      if (wrapperRef.current && !wrapperRef.current.contains(event.target)) {
        setShowResults(false);
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  const cargarPersona = async (id) => {
    try {
      const response = await api.get(`/personas/${id}/`);
      setSelectedPersona(response.data);
    } catch (error) {
      console.error('Error cargando persona:', error);
    }
  };

  const buscarPersonas = async (term) => {
    if (!term.trim()) {
      setResultados([]);
      return;
    }

    setLoading(true);
    try {
      const response = await api.get('/personas/', {
        params: { search: term }
      });
      setResultados(response.data);
    } catch (error) {
      console.error('Error buscando personas:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleSearchChange = (e) => {
    const term = e.target.value;
    setSearchTerm(term);
    buscarPersonas(term);
    setShowResults(true);
  };

  const handleSelectPersona = (persona) => {
    setSelectedPersona(persona);
    onChange(persona.id);
    setSearchTerm('');
    setResultados([]);
    setShowResults(false);
  };

  const handleClearSelection = () => {
    setSelectedPersona(null);
    onChange('');
    setSearchTerm('');
  };

  const handleCrearNueva = () => {
    console.log('🟢 Click en CREAR NUEVA PERSONA desde BuscadorPersona');
    console.log('Ejecutando onCrearNueva...');
    onCrearNueva();
    setShowResults(false);
  };

  return (
    <div className="relative" ref={wrapperRef}>
      {/* Persona seleccionada */}
      {selectedPersona ? (
        <div className="flex items-center justify-between p-2 bg-blue-50 dark:bg-blue-900/20 border border-blue-200 dark:border-blue-800 rounded-lg">
          <div>
            <p className="text-sm font-medium">
              {selectedPersona.apellido}, {selectedPersona.nombre}
            </p>
            <p className="text-xs text-gray-500">
              {selectedPersona.tipo_documento}: {selectedPersona.numero_documento}
            </p>
          </div>
          <button
            onClick={handleClearSelection}
            className="p-1 hover:text-red-500 transition-colors"
            title="Cambiar persona"
          >
            <X size={16} />
          </button>
        </div>
      ) : (
        /* Buscador */
        <div className="relative">
          <Search className="absolute left-2 top-1/2 transform -translate-y-1/2 text-gray-400" size={16} />
          <input
            type="text"
            value={searchTerm}
            onChange={handleSearchChange}
            onFocus={() => setShowResults(true)}
            placeholder="BUSCAR PERSONA POR NOMBRE O DOCUMENTO..."
            className="w-full pl-8 pr-3 py-1.5 text-sm rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-dark-elevated focus:ring-1 focus:ring-accent"
          />
          
          {/* Resultados de búsqueda */}
          {showResults && (searchTerm || resultados.length > 0) && (
            <div className="absolute z-50 w-full mt-1 bg-white dark:bg-dark-surface border border-gray-200 dark:border-gray-700 rounded-lg shadow-lg max-h-60 overflow-y-auto">
              {loading ? (
                <div className="p-3 text-center text-sm text-gray-500">
                  Buscando...
                </div>
              ) : resultados.length > 0 ? (
                <>
                  {resultados.map(persona => (
                    <button
                      key={persona.id}
                      onClick={() => handleSelectPersona(persona)}
                      className="w-full text-left p-2 hover:bg-gray-100 dark:hover:bg-gray-800 border-b border-gray-100 dark:border-gray-700 last:border-0"
                    >
                      <p className="text-sm font-medium">
                        {persona.apellido}, {persona.nombre}
                      </p>
                      <p className="text-xs text-gray-500">
                        {persona.tipo_documento}: {persona.numero_documento}
                      </p>
                    </button>
                  ))}
                  <div className="p-2 border-t border-gray-200 dark:border-gray-700">
                    <button
                      onClick={handleCrearNueva}
                      className="w-full flex items-center justify-center gap-1 p-1.5 text-xs bg-accent/10 hover:bg-accent/20 text-accent rounded-lg transition-colors"
                    >
                      <Plus size={14} />
                      CREAR NUEVA PERSONA
                    </button>
                  </div>
                </>
              ) : searchTerm ? (
                <div className="p-3 text-center">
                  <p className="text-sm text-gray-500 mb-2">No se encontraron personas</p>
                  <button
                    onClick={handleCrearNueva}
                    className="flex items-center justify-center gap-1 w-full p-2 text-xs bg-accent hover:bg-accent-hover text-white rounded-lg transition-colors"
                  >
                    <Plus size={14} />
                    CREAR "{searchTerm.toUpperCase()}"
                  </button>
                </div>
              ) : (
                <div className="p-3 text-center text-sm text-gray-500">
                  Escribí para buscar...
                </div>
              )}
            </div>
          )}
        </div>
      )}
    </div>
  );
};

export default BuscadorPersona;
