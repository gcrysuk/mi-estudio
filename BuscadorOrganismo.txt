import { useState, useEffect, useRef } from 'react';
import { Search, Plus, X } from 'lucide-react';
import api from '../../services/api';

const BuscadorOrganismo = ({ value, onChange, onCrearNuevo }) => {
  const [searchTerm, setSearchTerm] = useState('');
  const [resultados, setResultados] = useState([]);
  const [loading, setLoading] = useState(false);
  const [showResults, setShowResults] = useState(false);
  const [selectedOrganismo, setSelectedOrganismo] = useState(null);
  const wrapperRef = useRef(null);

  useEffect(() => {
    if (value && !selectedOrganismo) {
      cargarOrganismo(value);
    }
  }, [value]);

  useEffect(() => {
    const handleClickOutside = (event) => {
      if (wrapperRef.current && !wrapperRef.current.contains(event.target)) {
        setShowResults(false);
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  const cargarOrganismo = async (id) => {
    try {
      const response = await api.get(`/carpetas/organismo/${id}/`);
      setSelectedOrganismo(response.data);
    } catch (error) {
      console.error('Error cargando organismo:', error);
    }
  };

const buscarOrganismos = async (term) => {
  if (!term.trim()) {
    setResultados([]);
    return;
  }

  setLoading(true);
  try {
    console.log('🔍 Buscando organismos con término:', term);
    console.log('URL completa:', '/carpetas/organismos/?search=' + term);
    
    const response = await api.get('/carpetas/organismos/', {
      params: { search: term }
    });
    
    console.log('✅ Respuesta organismos:', response.data);
    setResultados(response.data);
  } catch (error) {
    console.error('❌ Error buscando organismos:', error);
    console.error('Status:', error.response?.status);
    console.error('Data:', error.response?.data);
    // Datos de ejemplo
    setResultados([
      { id: 1, nombre: 'JUZGADO CIVIL N°1' },
      { id: 2, nombre: 'JUZGADO LABORAL N°2' },
      { id: 3, nombre: 'CÁMARA DE APELACIONES' },
    ]);
  } finally {
    setLoading(false);
  }
};

  const handleSearchChange = (e) => {
    const term = e.target.value;
    setSearchTerm(term);
    buscarOrganismos(term);
    setShowResults(true);
  };

  const handleSelectOrganismo = (organismo) => {
    setSelectedOrganismo(organismo);
    onChange(organismo.id);
    setSearchTerm('');
    setResultados([]);
    setShowResults(false);
  };

  const handleClearSelection = () => {
    setSelectedOrganismo(null);
    onChange('');
    setSearchTerm('');
  };

  const handleCrearNuevo = () => {
    onCrearNuevo(searchTerm);
    setShowResults(false);
  };

  return (
    <div className="relative" ref={wrapperRef}>
      {selectedOrganismo ? (
        <div className="flex items-center justify-between p-2 bg-blue-50 dark:bg-blue-900/20 border border-blue-200 dark:border-blue-800 rounded-lg">
          <div>
            <p className="text-sm font-medium">{selectedOrganismo.nombre}</p>
          </div>
          <button
            onClick={handleClearSelection}
            className="p-1 hover:text-red-500 transition-colors"
            title="Cambiar organismo"
          >
            <X size={16} />
          </button>
        </div>
      ) : (
        <div className="relative">
          <Search className="absolute left-2 top-1/2 transform -translate-y-1/2 text-gray-400" size={16} />
          <input
            type="text"
            value={searchTerm}
            onChange={handleSearchChange}
            onFocus={() => setShowResults(true)}
            placeholder="BUSCAR ORGANISMO..."
            className="w-full pl-8 pr-3 py-1.5 text-sm rounded-lg border border-gray-300 dark:border-gray-600 bg-white dark:bg-dark-elevated focus:ring-1 focus:ring-accent"
          />
          
          {showResults && (searchTerm || resultados.length > 0) && (
            <div className="absolute z-50 w-full mt-1 bg-white dark:bg-dark-surface border border-gray-200 dark:border-gray-700 rounded-lg shadow-lg max-h-60 overflow-y-auto">
              {loading ? (
                <div className="p-3 text-center text-sm text-gray-500">Buscando...</div>
              ) : resultados.length > 0 ? (
                <>
                  {resultados.map(org => (
                    <button
                      key={org.id}
                      onClick={() => handleSelectOrganismo(org)}
                      className="w-full text-left p-2 hover:bg-gray-100 dark:hover:bg-gray-800 border-b border-gray-100 dark:border-gray-700"
                    >
                      <p className="text-sm font-medium">{org.nombre}</p>
                    </button>
                  ))}
                  <div className="p-2 border-t border-gray-200 dark:border-gray-700">
                    <button
                      onClick={handleCrearNuevo}
                      className="w-full flex items-center justify-center gap-1 p-1.5 text-xs bg-accent/10 hover:bg-accent/20 text-accent rounded-lg transition-colors"
                    >
                      <Plus size={14} />
                      CREAR NUEVO ORGANISMO
                    </button>
                  </div>
                </>
              ) : searchTerm ? (
                <div className="p-3 text-center">
                  <p className="text-sm text-gray-500 mb-2">No se encontraron organismos</p>
                  <button
                    onClick={handleCrearNuevo}
                    className="flex items-center justify-center gap-1 w-full p-2 text-xs bg-accent hover:bg-accent-hover text-white rounded-lg transition-colors"
                  >
                    <Plus size={14} />
                    CREAR "{searchTerm.toUpperCase()}"
                  </button>
                </div>
              ) : (
                <div className="p-3 text-center text-sm text-gray-500">Escribí para buscar...</div>
              )}
            </div>
          )}
        </div>
      )}
    </div>
  );
};

export default BuscadorOrganismo;
